  // Datos simulados como si vinieran de un scrapeo
  const products = [

    },
    {
      title: "Wing Duotone Slick 5m 2024 NUEVA",
      description: "Wing nueva sin uso. Modelo Slick ideal para principiantes e intermedios.",
      price: 1200000,
      currency: "ARS",
      platform: "Hardwind",
      image: "https://f-one.world/assets/images/kites/2024/WTF.jpg"
    },
    {
      title: "Tabla Windsurf JP Australia Freestyle Wave 95L",
      description: "Tabla de windsurf JP Australia modelo Freestyle Wave, ideal para freestyle y olas.",
      price: 450000,
      currency: "ARS",
      platform: "GPX Store",
      image: "https://jpaustralia.com/wp-content/themes/jp-theme/img/productimage3.jpg"
    }
  ];

  const productList = document.getElementById('productList');
  const platformFilter = document.getElementById('platformFilter');

  function renderProducts(productsToRender) {
    productList.innerHTML = '';
    productsToRender.forEach(product => {
      const card = document.createElement('div');
      card.className = 'card';
      card.innerHTML = `
        <img src="${product.image}" alt="${product.title}" />
        <div class="card-title">${product.title}</div>
        <div class="card-price">${product.price.toLocaleString()} ${product.currency}</div>
        <div class="card-platform">${product.platform}</div>
        <div class="card-description">${product.description}</div>
      `;
      productList.appendChild(card);
    });
  }

  platformFilter.addEventListener('change', () => {
    const selected = platformFilter.value;
    if (selected === 'all') {
      renderProducts(products);
    } else {
      const filtered = products.filter(p => p.platform === selected);
      renderProducts(filtered);
    }
  });

  // Render inicial
  renderProducts(products);
</script>
